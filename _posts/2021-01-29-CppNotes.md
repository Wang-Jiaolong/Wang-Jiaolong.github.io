    if ((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0))

### 是否为质数

```c++
bool is_zhi(int n) {
    for (int i = 2; i < n - 1; i++) {
        if (n % i == 0) {
            return 0;
        }
    }
    return 1;
}
```

### 是否为回文数

```c++
bool is_hui(int n) {
    unsigned i = n;
    unsigned m = 0;
    while (i > 0) {
        m = m * 10 + i % 10;
        i /= 10;
    }
    return m == n;
}
```



## 运算函数

### 阶乘

```c++
int fact(int n) {
    if (n > 1) {
        return n * fact(n-1);
    }
    return n;
}
```





## 打印函数

### 杨辉三角形

```c++
void printyanghui(int n) {
    static int d[20][21];
    d[0][1] = 1;
    for (int i = 1; i < n; i++) {
        for (int j = 1; j < i + 2; j++) {
            d[i][j] = d[i - 1][j - 1] + d[i - 1][j];
        }
    }
    for (int i = 0; i < n; i++) {
        for (int j = 1; j < i + 2; j++) {
            if (j == 1) {
                cout << d[i][j];
            } else {
                cout << " " << d[i][j];
            }
        }
        cout << endl;
    }
}
```
